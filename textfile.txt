    <script>
        function openTab(tabName) {
            // 1. コンテンツの切り替え
            let contents = document.getElementsByClassName("tab-content");
            for (let i = 0; i < contents.length; i++) {
                contents[i].classList.remove("active");
            }
            
            // 指定したIDのコンテンツが存在するか確認してから表示する
            let targetContent = document.getElementById(tabName);
            if (targetContent) {
                targetContent.classList.add("active");
            }

            // 2. タブの見た目の切り替え
            let tabLinks = document.querySelectorAll(".main-nav ul li a");
            for (let i = 0; i < tabLinks.length; i++) {
                tabLinks[i].classList.remove("active-tab");
            }
            
            // 指定したIDのタブが存在するか確認してから色を変える
            let targetTab = document.getElementById("nav-" + tabName);
            if (targetTab) {
                targetTab.classList.add("active-tab");
            }
            
            // 3. 画面上部にスクロール
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }
    </script>
